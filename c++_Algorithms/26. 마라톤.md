#include<iostream>
#include <vector>
#include<algorithm>
int main(){
	using namespace std;
	//freopen("input.txt", "rt", stdin);	
	int n,i, now, pre, pos;
	bool flag = true;
	
	//pre와 now를 이용하여 차값에 해당하는 배열의 위치를 1로만든다.
	vector <int>ch(n); //0으로 초기화되는 벡터 배열  
	cin>>n;
	cin>>pre;

	for(i=0; i<n; i++)
	{
		pos=abs(pre-now);
		cin>>now;
		if(pos>0 && pos<n && ch[pos]==0) ch[pos]=1;
		else
		{
			cout<<"NO";
			return 0;
		}
		pre=now;
	}
	cout<<"YES";
	return 0;
}// 배열을 이용하지 않은 pre, now를 이용하여 표현 , jolly number가 있는지만 계산한다.
