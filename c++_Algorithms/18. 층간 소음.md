
## 내가 짠 코드
```c++
#include<iostream>
int main(){
	using namespace std;
	//freopen("input.txt", "rt", stdin);
	
	int n,m,i,cnt=0,max=-2147000000,sensor;
	cin>>n>>m;
	for(i=0;i<n;i++){
		cin>>sensor;
		if(sensor>m){
			cnt++;
			if(max<=cnt){
			max=cnt; 
			}
		}else {
		cnt=0;
		}
	}
	if(max<0) cout<<"-1";
	else cout<<max<< "  ";
	
	return 0;	
}
```
