```c++
#include<iostream>
int a[101];
int main(){
	using namespace std;
	//freopen("input.txt", "rt", stdin);	
	int n,i;
	//경기수 n을 입력 받는다. 
	cin>>n;
	
	//하나의 배열에 B의 가위 바위 보 까지 입력 받는다. 
	for(i=0;i<2*n;i++){
		cin>>a[i];
	}
	
	//하나의 배열 안에서 5회까지 경기를 비교한다.
	//그 배열 값의 +1 한 것과 같다면 값이 높은 값을 가진 사람이 이긴다.
	// 만약 +1을 해도 같지 않다면 높은 값을 가진 사람이 진다. 
	for(i=0;i<n;i++){
		if(a[i] < a[n+i] ){
			if(a[i]+1 == a[n+i]) {
			cout<<"B"<<'\n';
			continue;
			}
			else {
			cout<<"A"<<'\n';
			continue;
			}
		}else if(a[i] > a[n+i]){
			if(a[n+i]+1 == a[i]){
			cout<< "A"<<'\n';
			continue;
			} 
			else{
			cout<<"B"<<'\n';
			continue;	
			} 
		}else cout<<"D"<<'\n'; continue;
	}
	return 0;	
}
```

## 풀이
#include<iostream>
int main(){
	using namespace std;
	//freopen("input.txt", "rt", stdin);	
	int n,i;
    int a[101],b[101];
	
    //경기수 n을 입력 받는다. 
	cin>>n;
	
	//각각의 배열에 가위 바위 보를 입력 받는다. 
	for(i=0;i<n;i++){
		cin>>a[i];
	}
	for(i=0;i<n;i++){
		cin>>b[i];
	}
    
    //A가 이기는 경우만 생각해서 경우를 나눠 놓으면 else에는 B가 
    //이기는 경우만 들어간다. 
    //굉장히 간단하게 생각 가능.
    for(i=0;i<n;i++){
        if(a[i]==b[i]) cout<<"D"<<'\n';
        else if (a[i]==1 && b[i]==3) cout<<"A\n";
		else if (a[i]==2 && b[i]==1) cout<<"A\n";    
		else if (a[i]==3 && b[i]==2) cout<<"A\n";
		else cout<<"B\n";
    }
	return 0;	
}
