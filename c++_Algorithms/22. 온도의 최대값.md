```c++
test 4, 5번에서 타임리미트 발생
시간복잡도 문제 때문에 타임 리미트가 발생함. 
#include<iostream>
int main(){
	using namespace std;
	//freopen("input.txt", "rt", stdin);	
	int i,j,n,k,sum=0;
	int max=-2147000000;
	
	cin>>n>>k;
	int temp[n];
	for(i=0;i<n;i++){
		cin>>temp[i];
	}
	
	for(i=0;i<n-k+1;i++){
		for(j=0;j<k;j++){
			sum+=temp[i+j];
		}
		
		if(max<sum)
			max=sum;
		sum=0; 
	}
	cout<<max;
}
```

```cpp
#include<iostream>
#include <vector>
int main(){
	using namespace std;
	//freopen("input.txt", "rt", stdin);	
	int i,n,k,sum=0;
	int max=-2147000000;
	
	cin>>n>>k;
	vector<int> a(n);
	//배열의 값의 0~9 처럼 잡힌다. 
	//초기값은 다 0으로 됨.
	 
	for(i=0;i<n;i++){
		cin>>a[i];
	}
	
	for(i=0;i<k;i++){
		sum+=a[i];
		
	}
	for(i=k; i<n; i++){
		sum=sum+(a[i]-a[i-k]);
		if(sum>max) max=sum;
	}
	cout<<max;
	
}
```
