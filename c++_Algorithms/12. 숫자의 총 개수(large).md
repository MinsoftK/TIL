
```c++
#include<stdio.h>
int main(){
	freopen("input.txt", "rt", stdin);
	int n, sum=0, cnt=1, digit=9, res=0;
	scanf("%d", &n);
	while(sum+digit<n){	
		sum=sum+digit;
		res=res+(cnt*digit);
		cnt++;
		digit=digit*10;
	}
	res=res+((n-sum)*cnt);
	printf("%d\n", res);
	return 0;
}
```

```c++
#include <iostream>

	using namespace std;

	cin>>value;
	res=value;
	while(sum+i<value){
		cnt++;
		sum=sum+i;
		sumres=sumres+cnt*i;
		i*=10;
	}
		res=sumres+(value-sum)*(cnt+1);
	cout<<res;
}
```
