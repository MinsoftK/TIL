```c++
#include<iostream>
#include <vector>
int main(){
	using namespace std;
//	freopen("input.txt", "rt", stdin);	
	int i,n,cnt=1,res;
	cin>>n;
	vector <int> a(n);
	for(i=0;i<n;i++){
		cin>>a[i];
	}
	for(i=0;i<n;i++){
		if(a[i]<=a[i+1]){
			cnt++;
			if(res<cnt)
				res=cnt;
		}else cnt=1;
	}
	cout<<res;
	return 0;
}
```
